import 'package:flutter_bloc/flutter_bloc.dart';

enum ContactEvent { start, success, failure }

class ContactBloc extends Bloc<ContactEvent, String> {
  ContactBloc() : super('');

  @override
  Stream<String> mapEventToState(ContactEvent event) async* {
    switch (event) {
      case ContactEvent.start:
        yield 'Loading...';
        break;
      case ContactEvent.success:
        yield 'Contact saved successfully!';
        break;
      case ContactEvent.failure:
        yield 'Failed to save contact.';
        break;
    }
  }
}
